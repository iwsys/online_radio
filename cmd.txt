git push origin master - запушить на удал сервер
git commit -m readme.txt - закоммитить файл
git commit -a -m "description" - закоммитить все файлы с описанием 
git rm - удалить файл с отлеживания
git log --pretty=format:"%h - %an, %ar : %s" - лог с коммитами
git commit --amend -m "Add an author/email comment" - изменить предыдущий коммит
git tag -d nametag -  удалить тег
